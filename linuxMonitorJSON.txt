#!/bin/bash
export LC_ALL=C
export LANG=C

# ─────────────────────────────────────────────
#  monitor-linux-json.sh - Linux System Monitor (JSON output)
#  Usage: bash monitor-linux-json.sh [-w] [-i SECONDS] [-s SECTIONS]
#    -w            Watch mode (auto-refresh)
#    -i SECONDS    Refresh interval (default: 3)
#    -s SECTIONS   Comma-separated sections to show:
#                  cpu, ram, disk, network, processes
#                  (default: all)
#
#  Examples:
#    bash monitor-linux-json.sh
#    bash monitor-linux-json.sh -s cpu,ram
#    bash monitor-linux-json.sh -w -i 5
#    bash monitor-linux-json.sh -s disk > stats.json
# ─────────────────────────────────────────────

INTERVAL=3
WATCH=false
SECTIONS="cpu,ram,disk,network,processes"

while getopts "wi:s:" opt; do
  case $opt in
    w) WATCH=true ;;
    i) INTERVAL="$OPTARG" ;;
    s) SECTIONS="$OPTARG" ;;
    *) echo "Usage: $0 [-w] [-i SECONDS] [-s SECTIONS]"; exit 1 ;;
  esac
done

has_section() {
  echo "$SECTIONS" | tr ',' '\n' | grep -qx "$1"
}

# ── JSON helpers ──────────────────────────────

json_str()  { printf '"%s"' "$1"; }
json_int()  { printf '%s'   "$1"; }
json_null() { printf 'null'; }

# ── Data collectors ───────────────────────────

get_header() {
  local hostname user uptime_sec uptime_str cores os_name timestamp

  hostname=$(hostname -s)
  user=$(whoami)
  uptime_sec=$(awk '{print int($1)}' /proc/uptime)
  uptime_str=$(uptime -p 2>/dev/null || uptime | sed 's/.*up //' | sed 's/,.*//')
  cores=$(nproc)
  os_name=$(grep PRETTY_NAME /etc/os-release 2>/dev/null | cut -d= -f2 | tr -d '"' || uname -s)
  timestamp=$(date '+%Y-%m-%dT%H:%M:%S')

  printf '  "header": {\n'
  printf '    "hostname":   %s,\n'   "$(json_str "$hostname")"
  printf '    "user":       %s,\n'   "$(json_str "$user")"
  printf '    "os":         %s,\n'   "$(json_str "$os_name")"
  printf '    "uptime_sec": %s,\n'   "$(json_int "$uptime_sec")"
  printf '    "uptime_str": %s,\n'   "$(json_str "$uptime_str")"
  printf '    "cores":      %s,\n'   "$(json_int "$cores")"
  printf '    "timestamp":  %s\n'    "$(json_str "$timestamp")"
  printf '  }'
}

get_cpu() {
  local stat1 stat2 idle1 idle2 total1 total2 used pct cores model

  stat1=$(grep '^cpu ' /proc/stat)
  sleep 0.5
  stat2=$(grep '^cpu ' /proc/stat)

  idle1=$(echo "$stat1" | awk '{print $5}')
  idle2=$(echo "$stat2" | awk '{print $5}')
  total1=$(echo "$stat1" | awk '{s=0; for(i=2;i<=NF;i++) s+=$i; print s}')
  total2=$(echo "$stat2" | awk '{s=0; for(i=2;i<=NF;i++) s+=$i; print s}')

  local diff_idle=$(( idle2 - idle1 ))
  local diff_total=$(( total2 - total1 ))
  pct=$(( (diff_total - diff_idle) * 100 / diff_total ))

  local user_pct=$(awk '{printf "%d", ($2-u)*100/(t2-t1)}' \
    u="$(echo "$stat1"|awk '{print $2}')" \
    t1="$total1" t2="$total2" <<< "$stat2")
  local sys_pct=$(awk '{printf "%d", ($4-s)*100/(t2-t1)}' \
    s="$(echo "$stat1"|awk '{print $4}')" \
    t1="$total1" t2="$total2" <<< "$stat2")

  cores=$(nproc)
  model=$(grep 'model name' /proc/cpuinfo | head -1 | cut -d: -f2 | xargs)

  local load_raw
  load_raw=$(cat /proc/loadavg)
  local l1=$(echo "$load_raw"  | awk '{print $1}')
  local l5=$(echo "$load_raw"  | awk '{print $2}')
  local l15=$(echo "$load_raw" | awk '{print $3}')

  printf '  "cpu": {\n'
  printf '    "model":      %s,\n'   "$(json_str "$model")"
  printf '    "cores":      %s,\n'   "$(json_int "$cores")"
  printf '    "load_pct":   %s,\n'   "$(json_int "$pct")"
  printf '    "user_pct":   %s,\n'   "$(json_int "$user_pct")"
  printf '    "system_pct": %s,\n'   "$(json_int "$sys_pct")"
  printf '    "idle_pct":   %s,\n'   "$(json_int $(( 100 - pct )))"
  printf '    "load_1m":    %s,\n'   "$(json_str "$l1")"
  printf '    "load_5m":    %s,\n'   "$(json_str "$l5")"
  printf '    "load_15m":   %s\n'    "$(json_str "$l15")"
  printf '  }'
}

get_ram() {
  local meminfo total free available buffers cached swap_total swap_free

  meminfo=$(cat /proc/meminfo)
  total=$(echo "$meminfo"      | awk '/^MemTotal:/     {print $2}')
  free=$(echo "$meminfo"       | awk '/^MemFree:/      {print $2}')
  available=$(echo "$meminfo"  | awk '/^MemAvailable:/ {print $2}')
  buffers=$(echo "$meminfo"    | awk '/^Buffers:/      {print $2}')
  cached=$(echo "$meminfo"     | awk '/^Cached:/       {print $2}')
  swap_total=$(echo "$meminfo" | awk '/^SwapTotal:/    {print $2}')
  swap_free=$(echo "$meminfo"  | awk '/^SwapFree:/     {print $2}')

  local used=$(( total - available ))
  local pct=$(( used * 100 / total ))
  local swap_used=$(( swap_total - swap_free ))
  local swap_pct=0
  (( swap_total > 0 )) && swap_pct=$(( swap_used * 100 / swap_total ))

  # Values in kB, convert to bytes
  printf '  "ram": {\n'
  printf '    "total_bytes":      %s,\n'  "$(json_int $(( total     * 1024 )))"
  printf '    "used_bytes":       %s,\n'  "$(json_int $(( used      * 1024 )))"
  printf '    "free_bytes":       %s,\n'  "$(json_int $(( free      * 1024 )))"
  printf '    "available_bytes":  %s,\n'  "$(json_int $(( available * 1024 )))"
  printf '    "buffers_bytes":    %s,\n'  "$(json_int $(( buffers   * 1024 )))"
  printf '    "cached_bytes":     %s,\n'  "$(json_int $(( cached    * 1024 )))"
  printf '    "used_pct":         %s,\n'  "$(json_int "$pct")"
  printf '    "swap_total_bytes": %s,\n'  "$(json_int $(( swap_total * 1024 )))"
  printf '    "swap_used_bytes":  %s,\n'  "$(json_int $(( swap_used  * 1024 )))"
  printf '    "swap_used_pct":    %s\n'   "$(json_int "$swap_pct")"
  printf '  }'
}

get_disk() {
  printf '  "disk": [\n'
  local first=true

  df -BM | grep -E '^/dev/' | while read -r line; do
    local dev=$(echo "$line"   | awk '{print $1}')
    local size=$(echo "$line"  | awk '{print $2}' | tr -d 'M')
    local used=$(echo "$line"  | awk '{print $3}' | tr -d 'M')
    local avail=$(echo "$line" | awk '{print $4}' | tr -d 'M')
    local pct=$(echo "$line"   | awk '{gsub(/%/,""); print $5}')
    local mount=$(echo "$line" | awk '{print $6}')

    [[ "$first" == true ]] && first=false || printf ',\n'

    printf '    {\n'
    printf '      "device":      %s,\n'  "$(json_str "$dev")"
    printf '      "mount":       %s,\n'  "$(json_str "$mount")"
    printf '      "total_bytes": %s,\n'  "$(json_int $(( size  * 1048576 )))"
    printf '      "used_bytes":  %s,\n'  "$(json_int $(( used  * 1048576 )))"
    printf '      "free_bytes":  %s,\n'  "$(json_int $(( avail * 1048576 )))"
    printf '      "used_pct":    %s\n'   "$(json_int "$pct")"
    printf '    }'
  done

  printf '\n  ]'
}

get_network() {
  local iface
  iface=$(ip route show default 2>/dev/null | awk '/default/{print $5; exit}')

  local ip_addr mac gateway rx_bytes tx_bytes

  if [[ -n "$iface" ]]; then
    ip_addr=$(ip addr show "$iface" 2>/dev/null | awk '/inet /{print $2}' | cut -d/ -f1 | head -1)
    mac=$(cat /sys/class/net/"$iface"/address 2>/dev/null)
    gateway=$(ip route show default | awk '{print $3}' | head -1)
    rx_bytes=$(cat /sys/class/net/"$iface"/statistics/rx_bytes 2>/dev/null || echo 0)
    tx_bytes=$(cat /sys/class/net/"$iface"/statistics/tx_bytes 2>/dev/null || echo 0)
  fi

  printf '  "network": {\n'
  printf '    "interface":   %s,\n'  "$(json_str "${iface:-unknown}")"
  printf '    "ip":          %s,\n'  "$(json_str "${ip_addr:-unknown}")"
  printf '    "mac":         %s,\n'  "$(json_str "${mac:-unknown}")"
  printf '    "gateway":     %s,\n'  "$(json_str "${gateway:-unknown}")"
  printf '    "rx_bytes":    %s,\n'  "$(json_int "${rx_bytes:-0}")"
  printf '    "tx_bytes":    %s\n'   "$(json_int "${tx_bytes:-0}")"
  printf '  }'
}

get_processes() {
  printf '  "processes": [\n'
  local first=true

  ps aux --sort=-%cpu | tail -n +2 | head -15 | while read -r line; do
    local pid=$(echo "$line"  | awk '{print $2}')
    local cpu=$(echo "$line"  | awk '{print $3}')
    local mem=$(echo "$line"  | awk '{print $4}')
    local user=$(echo "$line" | awk '{print $1}')
    local cmd=$(echo "$line"  | awk '{for(i=11;i<=NF;i++) printf "%s ", $i; print ""}' | sed 's|^[^ ]*/||' | cut -c1-60 | tr -d '"\\')

    [[ "$first" == true ]] && first=false || printf ',\n'

    printf '    {\n'
    printf '      "pid":      %s,\n'  "$(json_int  "$pid")"
    printf '      "cpu_pct":  %s,\n'  "$(json_str  "$cpu")"
    printf '      "mem_pct":  %s,\n'  "$(json_str  "$mem")"
    printf '      "user":     %s,\n'  "$(json_str  "$user")"
    printf '      "command":  %s\n'   "$(json_str  "$cmd")"
    printf '    }'
  done

  printf '\n  ]'
}

# ── Render ────────────────────────────────────

render() {
  local sections=()
  has_section "cpu"       && sections+=("cpu")
  has_section "ram"       && sections+=("ram")
  has_section "disk"      && sections+=("disk")
  has_section "network"   && sections+=("network")
  has_section "processes" && sections+=("processes")

  printf '{\n'
  get_header

  for sec in "${sections[@]}"; do
    printf ',\n'
    case $sec in
      cpu)       get_cpu ;;
      ram)       get_ram ;;
      disk)      get_disk ;;
      network)   get_network ;;
      processes) get_processes ;;
    esac
  done

  printf '\n}\n'
}

if $WATCH; then
  while true; do
    clear
    render
    sleep "$INTERVAL"
  done
else
  render
fi
